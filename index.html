Customer Churn Analysis Query


SELECT contract_type,
       COUNT(*) AS total_customers,
       SUM(CASE WHEN churn = 'Yes' THEN 1 ELSE 0 END) AS churned_customers
FROM telecom_data
GROUP BY contract_type
ORDER BY churned_customers DESC;
